def rank_sort(x):
        x_order = np.argsort(x)
        x_rank = np.zeros(len(x), int)
        j = 0
        for i in x_order:
                x_rank[i] = j
                j = j + 1
        return x_rank

def poly_redu(a):
    a = list(a)  # Force it to be a list.
    while np.abs(a[0]) < eps:
        a = np.delete(a, 0)
    return a  # return a list

def poly_add(*poly):
    n = len(poly)
    if n == 1:
        c = poly[0]
        c = poly_redu(c)
    elif n == 2:
        c = np.polyadd(poly[0], poly[1])
        c = poly_redu(c)
    else:
        c = np.polyadd(poly[0], poly[1])
        c = poly_redu(c)
        for i in range(2, n):
            c = np.polyadd(c, poly[i])
            c = poly_redu(c)
    return c
